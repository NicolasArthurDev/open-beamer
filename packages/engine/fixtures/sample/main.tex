% open-beamer core fixture (the blocking gate — compiles on any TeX Live with beamer+tikz).
% Exercises the genuinely risky pieces of the pipeline: TikZ overlay with absolute
% positioning, `remember picture`/`calc`, and the 2nd pass those references force, plus
% real PDF output. Font handling (fontspec) is covered separately in ../fontspec.
% Compile with lualatex (or pdflatex/xelatex), 2 passes.

\usepackage{tikz}
\usetikzlibrary{calc}
\definecolor{obAccent}{HTML}{1A365D}

\end{document}
